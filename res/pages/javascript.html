	function checkBrowserIE()
	{
	document.getElementById("dataDump").innerHTML = "java = " + navigator.javaEnabled();
	if (navigator.javaEnabled() == true)
	{
		document.getElementById("dataDump").innerHTML += "edge";
		var isIE = String(navigator.appVersion);
		document.getElementById("dataDump").innerHTML += isIE;
		var detect = isIE.indexOf("Trident");
		document.getElementById("dataDump").innerHTML += detect;
		
		if (detect > 0)
		{
			document.getElementById("dataDump").innerHTML = "IE";
		}
		else
		{
			document.getElementById("dataDump").innerHTML = "EDGE";
		}
		
	}
	else
	{
		document.getElementById("dataDump").innerHTML += "CHROME";
	}
	
	var browserVER = document.getElementById("dataDump").innerHTML;
	if (browserVER == "IE")
		{
		location.href = "index_IE.html";
		}
	if (browserVER == "EDGE")
		{
		location.href = "index_EDGE.html";
		}
	}

	function expandFunction()
{
	var prevButton = document.getElementById("dataDump").innerHTML;	
	var clickedButton = document.activeElement.innerHTML.split("()").slice(0,-1);
	var clickedButton = String(clickedButton).substr(19);
	document.getElementById("dataDump").innerHTML = clickedButton;
	var isShowFunction = document.getElementById(clickedButton);

	if (prevButton == clickedButton || prevButton == "")
	{	
		var showFunction = document.getElementById(clickedButton);	
		showFunction.classList.toggle("functionContent_show_" + clickedButton);
		document.getElementById("if status").innerHTML = "if triggered";
	}
	else
	{
		var showFunction = document.getElementById(prevButton);	
		showFunction.classList.toggle("functionContent_show_" + prevButton);
		var showFunction = document.getElementById(clickedButton);	
		showFunction.classList.toggle("functionContent_show_" + clickedButton);		
	}
}
